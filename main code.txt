#include <SPI.h>
#include <MFRC522.h>
#include <Servo.h>
#include <LiquidCrystal_I2C.h>
#include <dht.h>


LiquidCrystal_I2C lcd(0x27,16,2);


#define SS_PIN 53
#define RST_PIN 48

Servo Servo1;
Servo Servo2;
int pos = 0; 


String tagId = "";
String MasterTag = "63B6C72E";

uint8_t BlueLightPin = 11;
uint8_t RedLightPin = 10;
uint8_t BuzzerPin = 9;
uint8_t ServoPin1 = 12;
uint8_t ServoPin2 = 13;
uint8_t DHT_Pin = 8;

dht DHT11;

MFRC522 mfrc522(SS_PIN, RST_PIN);
// MFRC522::MIFARE_key key;

void setup() {
  pinMode(BlueLightPin, OUTPUT);  
  pinMode(RedLightPin, OUTPUT);  
  pinMode(BuzzerPin, OUTPUT);  

  Servo1.attach(ServoPin1);
  Servo2.attach(ServoPin2);
  Servo1.write(pos);              // tell servo to go to position in variable 'pos'
  Servo2.write(pos);              // tell servo to go to position in variable 'pos'


  digitalWrite(BlueLightPin, HIGH);
  digitalWrite(RedLightPin, HIGH);
  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init();
  Serial.println("Scan you RFID card");

  lcd.init();
  lcd.clear();         
  lcd.backlight(); 

  lcd.setCursor(0,0); 
  lcd.print("Welcome !");

}


void temp_sensor(){
  lcd.clear();
  lcd.setCursor(0,0); 
  lcd.print("Temp = ");
  lcd.print(DHT11.temperature);

  lcd.setCursor(0,1); 
  lcd.print("Hdt = ");
  lcd.print(DHT11.humidity);
  lcd.print("%");

  delay(2000);
}

void loop() {
  int chk = DHT11.read11(DHT_Pin);
  Serial.println(chk);
  if(chk){
    temp_sensor();
  }
  
  


  pos = 0;
  Servo1.write(pos);  
  if( ! mfrc522.PICC_IsNewCardPresent()){
    return;
  }

  if( ! mfrc522.PICC_ReadCardSerial()){
    return;
  }

  tagId = "";
  for(int i=0; i<4; i++){
    tagId.concat(String(mfrc522.uid.uidByte[i], HEX));
  }
  tagId.toUpperCase();
  Serial.println(tagId);

  if(tagId == MasterTag){
    Serial.println("Matched");
    digitalWrite(RedLightPin, HIGH);
    digitalWrite(BlueLightPin, LOW);
    tone(BuzzerPin, 1000);
    delay(80);
    noTone(BuzzerPin);
    delay(40);
    tone(BuzzerPin, 1000);
    delay(80);
    noTone(BuzzerPin);

    lcd.clear();  
    lcd.setCursor(0,0);   //Set cursor to character 2 on line 0
    lcd.print("Opening Visiting");
    lcd.setCursor(0,1);   //Set cursor to character 2 on line 0
    lcd.print("Door");

    Servo1.write(90);  
    

    for (pos = 0; pos <= 90; pos += 3) { // goes from 0 degrees to 180 degrees
    // in steps of 1 degree
      Servo1.write(pos);              // tell servo to go to position in variable 'pos'
      Serial.println(pos);
      delay(15);                       // waits 15ms for the servo to reach the position
    }
    delay(800);
    digitalWrite(BlueLightPin, HIGH);
    digitalWrite(RedLightPin, HIGH);
    delay(1000);
    for (pos = 90; pos >= 0; pos -= 3) { // goes from 180 degrees to 0 degrees
      Serial.println(pos);
      Servo1.write(pos);              // tell servo to go to position in variable 'pos'
      delay(15);                       // waits 15ms for the servo to reach the position
    }
     
    


  }
  else{
    Serial.println("Mismatched");
    digitalWrite(BlueLightPin, HIGH);
    digitalWrite(RedLightPin, LOW);
    tone(BuzzerPin, 1000);
    delay(500);
    noTone(BuzzerPin);


    lcd.clear();  
    lcd.setCursor(0,0);   //Set cursor to character 2 on line 0
    lcd.print("Opening Staff");
    lcd.setCursor(0,1);   //Set cursor to character 2 on line 0
    lcd.print("Door");


    Serial.println(pos);
    Servo2.write(95);   
    delay(800);
    digitalWrite(BlueLightPin, HIGH);
    digitalWrite(RedLightPin, HIGH);
    delay(1500);
    for (pos = 90; pos >= 0; pos -= 10) { // goes from 180 degrees to 0 degrees
    Servo2.write(pos);              // tell servo to go to position in variable 'pos'
    delay(10);                       // waits 15ms for the servo to reach the position
    }
  }
  mfrc522.PICC_HaltA();

  

}
